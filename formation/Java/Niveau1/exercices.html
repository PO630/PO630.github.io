Le Juste Prix

Dans cet exercice, vous allez créer un jeu de "Juste Prix".

  • L'application choisira un nombre aléatoire entre 0 et 100.
  • Le joueur devra deviner ce nombre en un maximum de 5 tentatives.
  • Après chaque essai, l'application indiquera si le nombre donné est plus grand ou plus petit que le nombre mystère.
  • À la fin du jeu, le joueur pourra décider de rejouer ou de quitter l'application.

Générer un Nombre Mystère

Écrire un programme qui génère un nombre aléatoire entre 0 et 100.


    import java.util.Random;

    public class JustePrix {
        public static void main(String[] args) {
            Random random = new Random();
            int nombreMystere = random.nextInt(101); // Nombre entre 0 et 100
            System.out.println("Le nombre mystère est : " + nombreMystere);
        }
    }

Permettre au Joueur de Proposer un Nombre

Ajoutez un mécanisme permettant au joueur de saisir un nombre et affichez ce qu'il a entré.


    import java.util.Scanner;

    public class JustePrix {
        public static void main(String[] args) {
            Scanner scanner = new Scanner(System.in);
            System.out.print("Proposez un nombre : ");
            int proposition = scanner.nextInt();
            System.out.println("Vous avez proposé : " + proposition);
        }
    }

Comparer la Proposition avec le Nombre Mystère

Comparez le nombre proposé avec le nombre mystère et indiquez si c'est plus grand, plus petit ou correct.


    import java.util.Random;
    import java.util.Scanner;
    
    public class JustePrix {
        public static void main(String[] args) {
            Random random = new Random();
            int nombreMystere = random.nextInt(101); // Nombre entre 0 et 100
    
            Scanner scanner = new Scanner(System.in);
            System.out.print("Proposez un nombre : ");
            int proposition = scanner.nextInt();
    
            if (proposition == nombreMystere) {
                System.out.println("Bravo, vous avez trouvé le nombre mystère !");
            } else if (proposition < nombreMystere) {
                System.out.println("C'est plus !");
            } else {
                System.out.println("C'est moins !");
            }
        }
    }

Ajouter un Nombre Limité de Tentatives

Limitez le joueur à 5 tentatives et affichez un message s’il échoue.


    import java.util.Random;
    import java.util.Scanner;
    
    public class JustePrix {
        public static void main(String[] args) {
            Random random = new Random();
            int nombreMystere = random.nextInt(101); // Nombre entre 0 et 100
            Scanner scanner = new Scanner(System.in);
    
            int tentatives = 5;
            boolean gagne = false;
    
            while (tentatives > 0) {
                System.out.print("Proposez un nombre (Tentatives restantes : " + tentatives + ") : ");
                int proposition = scanner.nextInt();
                tentatives--;
    
                if (proposition == nombreMystere) {
                    System.out.println("Bravo, vous avez trouvé le nombre mystère !");
                    gagne = true;
                    break;
                } else if (proposition < nombreMystere) {
                    System.out.println("C'est plus !");
                } else {
                    System.out.println("C'est moins !");
                }
            }
    
            if (!gagne) {
                System.out.println("Dommage, vous avez perdu. Le nombre mystère était : " + nombreMystere);
            }
        }
    }

Ajouter la Possibilité de Rejouer

Permettez au joueur de rejouer ou de quitter après chaque partie.


    import java.util.Random;
    import java.util.Scanner;
    
    public class JustePrix {
        public static void main(String[] args) {
            Scanner scanner = new Scanner(System.in);
    
            boolean rejouer;
            do {
                Random random = new Random();
                int nombreMystere = random.nextInt(101); // Nombre entre 0 et 100
    
                int tentatives = 5;
                boolean gagne = false;
    
                while (tentatives > 0) {
                    System.out.print("Proposez un nombre (Tentatives restantes : " + tentatives + ") : ");
                    int proposition = scanner.nextInt();
                    tentatives--;
    
                    if (proposition == nombreMystere) {
                        System.out.println("Bravo, vous avez trouvé le nombre mystère !");
                        gagne = true;
                        break;
                    } else if (proposition < nombreMystere) {
                        System.out.println("C'est plus !");
                    } else {
                        System.out.println("C'est moins !");
                    }
                }
    
                if (!gagne) {
                    System.out.println("Dommage, vous avez perdu. Le nombre mystère était : " + nombreMystere);
                }
    
                System.out.print("Voulez-vous rejouer ? (oui/non) : ");
                String reponse = scanner.next();
                rejouer = reponse.equalsIgnoreCase("oui");
    
            } while (rejouer);
    
            System.out.println("Merci d'avoir joué. À bientôt !");
        }
    }
